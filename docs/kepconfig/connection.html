302    def _config_add(self, url, DATA):
303        '''Conducts an POST method at *url* to add an object in the Kepware Configuration
304        *DATA* is required to be a properly JSON object (dict) of the item to be posted to *url* 
305        '''
306        if len(DATA) == 0:
307            err_msg = f'Error: Empty List or Dict in DATA | DATA type: {type(DATA)}'
308            raise KepError(err_msg) 
309        data = json.dumps(DATA).encode('utf-8')
310        url_obj = self.__url_validate(url)
311        q = request.Request(url_obj, data, method='POST')
312        r = self.__connect(q)
313        return r

323    #Function used to Update an object to Kepware (HTTP PUT)
324    def _config_update(self, url, DATA = None):
325        '''Conducts an PUT method at *url* to modify an object in the Kepware Configuration.
326        *DATA* is required to be a properly JSON object (dict) of the item to be put to *url*
327        '''
328        url_obj = self.__url_validate(url)
329        if DATA == None:            
330            q = request.Request(url_obj, method='PUT')
331        else:
332            data = json.dumps(DATA).encode('utf-8')
333            q = request.Request(url_obj, data, method='PUT')
334        r = self.__connect(q)
335        return r

337    #Function used to Read an object from Kepware (HTTP GET) and return the JSON response
338    def _config_get(self, url, *, params = None):
339        '''
340        Conducts an GET method at *url* to retrieve an objects properties with query parameters in 
341        the Kepware Configuration.
342        '''
343        # Add parameters when necessary
344        if params is not None and params != {}:
345            qparams = parse.urlencode(params)
346            url = f'{url}?{qparams}'
347        url_obj = self.__url_validate(url)
348        q = request.Request(url_obj, method='GET')
349        r = self.__connect(q)
350        return r

353    def _force_update_check(self, force, DATA):
354        '''
355        This will validate if the modify call needs to be forced or not. If forced, the dict DATA needs
356        to have the 'FORCE_UPDATE' property with a value of True. If forced is not requested, it is necessary
357        to provide the current 'PROJECT_ID'. If 'PROJECT_ID' is not present in DATA, this will automatically 
358        retreive it from the active server.
359        '''
360        if force == True:
361            DATA['FORCE_UPDATE'] = True
362        else:
363            # Get Project ID if it doesn't exist and if FORCE_UPDATE is existing and FALSE
364            if 'PROJECT_ID' not in DATA:
365                if 'FORCE_UPDATE' in DATA:
366                    if 'FORCE_UPDATE' == False:
367                        try:
368                            project_data = self._config_get(self.url + '/project')
369                            DATA['PROJECT_ID'] = project_data.payload['PROJECT_ID']
370                        except:
371                            #NEED TO COVER ERROR CONDITION
372                            pass
373                else:
374                    try:
375                        project_data = self._config_get(self.url + '/project')
376                        DATA['PROJECT_ID'] = project_data.payload['PROJECT_ID']
377                    except:
378                        #NEED TO COVER ERROR CONDITION
379                        pass
380        return DATA
381    # General service call handler
382    def _kep_service_execute(self, url, DATA = None, TTL = None):
383        try:
384            if TTL != None:
385                if DATA == None: DATA = {}
386                DATA["servermain.JOB_TIME_TO_LIVE_SECONDS"]= TTL
387            r = self._config_update(url, DATA)
388            job = KepServiceResponse(r.payload['code'],r.payload['message'], r.payload['href'])
389            return job
390        except KepHTTPError as err:
391            if err.code == 429:
392                job = KepServiceResponse()
393                job.code = err.code
394                job.message = err.payload
395                return job
396            else:
397                raise err

403    # General connect call to manage HTTP responses for all methods
404    # Returns the response object for the method to handle as appropriate
405    # Raises Errors as found
406    def __connect(self,request_obj):
407        # Fill appropriate header information
408        data = _HttpDataAbstract()
409        request_obj.add_header("Authorization", "Basic %s" % self.__build_auth_str(self.username, self.password))
410        request_obj.add_header("Content-Type", "application/json")
411        request_obj.add_header("Accept", "application/json")
412        try:
413            # context is sent regardless of HTTP or HTTPS - seems to be ignored if HTTP URL
414            with request.urlopen(request_obj, context=self.__ssl_context) as server:
415                try:
416                    payload = server.read()
417                    data.payload = json.loads(codecs.decode(payload,'utf-8-sig'))
418                except:
419                    pass
420                data.code = server.code
421                data.reason = server.reason
422                return data
423        except error.HTTPError as err:
424            payload = json.loads(codecs.decode(err.read(),'utf-8-sig'))
425            # print('HTTP Code: {}\n{}'.format(err.code,payload), file=sys.stderr)
426            raise KepHTTPError(url=err.url, code=err.code, msg=err.msg, hdrs=err.hdrs, payload=payload)
427        except error.URLError as err:
428            # print('URLError: {} URL: {}'.format(err.reason, request_obj.get_full_url()), file=sys.stderr)
429            raise KepURLError(msg=err.reason, url=request_obj.get_full_url())

431    # Fucntion used to ensure special characters are handled in the URL
432    # Ex: Space will be turned to %20
433    def __url_validate(self, url):
434        # Configuration API does not use fragments in URL so ignore to allow # as a character
435        # Objects in Kepware can include # as part of the object names
436        parsed_url = parse.urlparse(url, allow_fragments= False)
437        # Added % for scenarios where special characters have already been escaped with %
438        updated_path = parse.quote(parsed_url.path, safe = '/%')
439
440        # If host is "localhost", force using the IPv4 loopback adapter IP address in all calls
441        # This is done to remove retries that will happen when the host resolution uses IPv6 intially
442        # Kepware currently doesn't support IPv6 and is not listening on this interface
443        if parsed_url.hostname.lower() == 'localhost':
444            ip = socket.gethostbyname(parsed_url.hostname)
445            parsed_url = parsed_url._replace(netloc='{}:{}'.format(ip, parsed_url.port))
446        
447        return parsed_url._replace(path=updated_path).geturl()

449    # Function used to build the basic authentication string
450    def __build_auth_str(self, username, password):
451        if isinstance(username, str):
452            username = username.encode('latin1')
453        if isinstance(password, str):
454            password = password.encode('latin1')
455        authstr = b64encode(b':'.join((username, password))).strip().decode('ascii')
456        return authstr

458    # Create parameters for log queries
459    def __create_query(self, start = None, end = None, limit = None):
460        query = {}
461        if start != None and isinstance(start, datetime.datetime):
462            query['start'] = start.isoformat()
463        if end != None and isinstance(end, datetime.datetime):
464            query['end'] = end.isoformat()
465        if limit != None:
466            query['limit'] = limit
467        return query

354    def _force_update_check(self, force, DATA):
355        '''
356        This will validate if the modify call needs to be forced or not. If forced, the dict DATA needs
357        to have the 'FORCE_UPDATE' property with a value of True. If forced is not requested, it is necessary
358        to provide the current 'PROJECT_ID'. If 'PROJECT_ID' is not present in DATA, this will automatically 
359        retreive it from the active server.
360        '''
361        if force == True:
362            DATA['FORCE_UPDATE'] = True
363        else:
364            # Get Project ID if it doesn't exist and if FORCE_UPDATE is existing and FALSE
365            if 'PROJECT_ID' not in DATA:
366                if 'FORCE_UPDATE' in DATA:
367                    if 'FORCE_UPDATE' == False:
368                        try:
369                            project_data = self._config_get(self.url + '/project')
370                            DATA['PROJECT_ID'] = project_data.payload['PROJECT_ID']
371                        except:
372                            #NEED TO COVER ERROR CONDITION
373                            pass
374                else:
375                    try:
376                        project_data = self._config_get(self.url + '/project')
377                        DATA['PROJECT_ID'] = project_data.payload['PROJECT_ID']
378                    except:
379                        #NEED TO COVER ERROR CONDITION
380                        pass
381        return DATA

405    # Returns the response object for the method to handle as appropriate
406    # Raises Errors as found
407    def __connect(self,request_obj):
408        # Fill appropriate header information
409        data = _HttpDataAbstract()
410        request_obj.add_header("Authorization", "Basic %s" % self.__build_auth_str(self.username, self.password))
411        request_obj.add_header("Content-Type", "application/json")
412        request_obj.add_header("Accept", "application/json")
413        try:
414            # context is sent regardless of HTTP or HTTPS - seems to be ignored if HTTP URL
415            with request.urlopen(request_obj, context=self.__ssl_context) as server:
416                try:
417                    payload = server.read()
418                    data.payload = json.loads(codecs.decode(payload,'utf-8-sig'))
419                except:
420                    pass
421                data.code = server.code
422                data.reason = server.reason
423                return data
424        except error.HTTPError as err:
425            payload = json.loads(codecs.decode(err.read(),'utf-8-sig'))
426            # print('HTTP Code: {}\n{}'.format(err.code,payload), file=sys.stderr)
427            raise KepHTTPError(url=err.url, code=err.code, msg=err.msg, hdrs=err.hdrs, payload=payload)
428        except error.URLError as err:
429            # print('URLError: {} URL: {}'.format(err.reason, request_obj.get_full_url()), file=sys.stderr)
430            raise KepURLError(msg=err.reason, url=request_obj.get_full_url())

432    # Fucntion used to ensure special characters are handled in the URL
433    # Ex: Space will be turned to %20
434    def __url_validate(self, url):
435        # Configuration API does not use fragments in URL so ignore to allow # as a character
436        # Objects in Kepware can include # as part of the object names
437        parsed_url = parse.urlparse(url, allow_fragments= False)
438        # Added % for scenarios where special characters have already been escaped with %
439        updated_path = parse.quote(parsed_url.path, safe = '/%')
440
441        # If host is "localhost", force using the IPv4 loopback adapter IP address in all calls
442        # This is done to remove retries that will happen when the host resolution uses IPv6 intially
443        # Kepware currently doesn't support IPv6 and is not listening on this interface
444        if parsed_url.hostname.lower() == 'localhost':
445            ip = socket.gethostbyname(parsed_url.hostname)
446            parsed_url = parsed_url._replace(netloc='{}:{}'.format(ip, parsed_url.port))
447        
448        return parsed_url._replace(path=updated_path).geturl()

452        if isinstance(username, str):
453            username = username.encode('latin1')
